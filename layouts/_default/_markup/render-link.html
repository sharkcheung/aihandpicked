{{- /* Render link hook: prepend language prefix for internal links */ -}} {{- $dest := .Destination -}} {{- if hasPrefix $dest "/" -}} {{- $prefix := "" -}} {{- if ne (site.Language.Lang) "en" -}} {{- $prefix = printf "/%s" (site.Language.Lang) -}} {{- end -}} {{- $dest = printf "%s%s" $prefix $dest -}} {{- end -}} {{ .Text }}
